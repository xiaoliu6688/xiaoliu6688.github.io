AXI-Lite总线设计

采用AXI-Lite协议，只支持单次传输，不支持突发传输，接口信号简单。由于dcache既需要读操作，也需要写操作，以dcache为例进行说明；

读操作

读操作的依赖关系：先完成读地址通道的握手，再进行读数据的握手，得到读数据。

  • master
// AR
assign axi_araddr = (state == RADDR)? &#123;addr[63:3],3'b000&#125; : 64'h0;
assign axi_arvalid = (state == RADDR)? 1 : 0;
// R
assign axi_rready = (state == RDATA)? 1 : 0;
  • slave
// read operation
always @(posedge clk or negedge rstn) begin
  if(!rstn) 
    rvalid <= 0;
  else if(rvalid & rready)
    rvalid <= 0;
  else if(arvalid & arready)
    rvalid <= 1;
end

reg flag;
always @(posedge clk or negedge rstn) begin
  if(!rstn)
    flag <= 0;
  else if(arvalid)
    flag <= 1;
  else
    flag <= 0;
end

reg [3:0] counter;
always @(posedge clk or negedge rstn) begin
  if(!rstn)
    counter <= 'h0;
  else if(flag)
    counter <= counter + 'h1;
  else
    counter <= 'h0;
end

always @(posedge clk or negedge rstn) begin
  if(!rstn) 
    arready <= 0; 
  else if(arvalid & arready)
    arready <= 0;
  else if(counter == 4)
    arready <= 1;
end

always @(posedge clk or negedge rstn) begin
  if(!rstn)
    data = 64'h0;
  else if(arvalid)
    data = paddr_read(araddr, 8);
  else 
    data = 'h0;
end

assign rdata = rvalid ? data : 'h0;
// counter 用来模拟salve提供数据所需要的周期数，延迟握手。

写操作

写操作中，写地址和写数据通道没有明确的握手依赖关系，但是写响应通道需要写地址和写数据握手后才可以握手，由于AXI-Lite只支持单次数据传输，所以没有wlast参与。本设计中，先进行写地址握手，再进行写数据握手，最后写响应通道握手，完成写操作。

  • master
// AW
assign axi_awaddr  = (state == WADDR) ? addr : 64'h0;
assign axi_awvalid = (state == WADDR) ? 1 : 0;
// W
assign axi_wdata  = (state == WDATA) ? wdata : 64'h0;
assign axi_wstrb  = (flash_flag&dirty) ? 8'b11111111 : (flash_flag) ? 8'b0000_0000 : byte_enable;
assign axi_wvalid = (state == WDATA) ? 1 : 0;
// B
assign axi_bready = (state == WRESP) ? 1 : 0;
  • slave
// write operation
reg [63:0] waddr;
always @(posedge clk or negedge rstn) begin
  if(!rstn) 
    waddr <= 'h0;
  else if(awvalid)
    waddr <= awaddr;
end

reg [63:0] wdata_r;
always @(posedge clk or negedge rstn) begin
  if(!rstn) 
    wdata_r <= 'h0;
  else if(wvalid)
    wdata_r <= wdata;
end

always @(posedge clk or negedge rstn) begin
  if(!rstn)
    awready <= 0;
  else if(awvalid && awready)
    awready <= 0;
  else if(awvalid)
    awready <= 1;
end

always @(posedge clk or negedge rstn) begin
  if(!rstn)
    wready <= 0;
  else if(wvalid && wready)
    wready <= 0;
  else if(wvalid)
    wready <= 1;
end

always @(posedge clk or negedge rstn) begin
  if(wvalid && wready) begin
    $display("write addr:%x, data:%x", waddr, wdata_r);
    case(wstrb) 
      8'b00000001: paddr_write(waddr, 1, wdata_r);
      8'b00000011: paddr_write(waddr, 2, wdata_r);
      8'b00001111: paddr_write(waddr, 4, wdata_r);
      8'b11111111: paddr_write(waddr, 8, wdata_r);
    endcase
  end
end

always @(posedge clk or negedge rstn) begin
  if(!rstn)
    bvalid <= 0;
  else if(bvalid && bready)
    bvalid <= 0;
  else if(wready & wvalid)
    bvalid <= 1;
end

assign bresp = bvalid ? 2'b00 : 2'b00;

AXI仲裁器

SRAM一般只有一个，Icache和Dcache可能出现同时访问SRAM的情况，产生了冲突，需要设计Arbiter来决定哪个master的优先级更高。当发生仲裁时，本设计中设计Dcache优先级更高。

Icache没有写操作的需要，不需要仲裁。只有Icache和Dcache同时发生读操作时才进行仲裁。

always @(*) begin
  if(lsu_arvalid) begin
    arvalid = lsu_arvalid;
    araddr  = lsu_araddr;
  end else begin
    arvalid = if_arvalid;
    araddr  = if_araddr;
  end
end
always @(*) begin
  if(lsu_arvalid)
    if_arready = 0;
  else 
    if_arready = arready;
end  
always @(*) begin
  if(lsu_arvalid)
    if_rvalid = 0;
  else 
    if_rvalid = rvalid;
end

// 当lsu_arvalid为1时，反压if_arready和if_rvalid，延迟Icache读地址通道和读数据通道的握手
