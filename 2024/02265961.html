握手协议

主要有三种握手场景

  • valid先拉高
  • ready先拉高
  • valid、ready同时拉高

需要注意：

  1. valid和ready信号是独立的，没有明确的依赖关系。（但是在AXI协议中，ready可以依赖valid拉高再拉高，反之，可能会产生死锁）
  2. valid信号需要和有效数据对齐
  3. 握手成功后，数据需要更新；如果没有新的数据更新，需要将valid拉低

握手接口信号

说明：一般来说握手的具体描述在master和slave内部，但是可以独立出来作为单独的模块，握手在该模块中完成；（在master端仍然需要一次握手，目的是修改data_i的数据；在salve端不需要握手，ready_i的更新只依赖于slave自身行为）

握手模块内部采用时序逻辑，将握手时的数据寄存一拍给salve，握手后，将握手的结果valid_o和data_o给slave端，这时valid_o相当于data_i的使能信号，不能当做一般的握手信号了。

// 中间一级寄存器
module handshake(
    input       clk,
    input       nrst,

    input       valid_i,
    input [7:0] data_i,
    output      ready_o,

    output      valid_o,
    output [7:0]data_o,
    input       ready_i 
);

reg [7:0] data_o_r;
reg valid_o_r;
always @(posedge clk or negedge nrst) begin
    if(!nrst)
        valid_o_r <= 0;
    else if(valid_i & ready_o)
        valid_o_r <= valid_i;
    else 
        valid_o_r <= 0;
end

always @(posedge clk) begin
    if(valid_i & ready_o)
        data_o_r <= data_i;
end

assign valid_o = valid_o_r;
assign data_o = data_o_r;
assign ready_o = ready_i & valid_o;

endmodule

具有两级缓冲的带握手功能的寄存器，可以用深度为2的FIFO代替。

// two stage register
module handshake(
    input           clk,
    input           nrst,
    input           valid_i,
    input [31:0]    data_i,
    output          ready_o,

    output          valid_o,
    output [31:0]   data_o,
    input           ready_i
); 

reg [31:0] slice_reg, spill_reg;
reg slice_full, spill_full;

always @(posedge clk or negedge nrst) begin
    if(!nrst)
        slice_reg <= 0;
    else if(!spill_full & valid_i)
        slice_reg <= data_i;
end 

always @(posedge clk or negedge nrst) begin
    if(!nrst)
        slice_full <= 0;
    else if(!spill_full)
        slice_full <= valid_i;
end 

always @(posedge clk or negedge nrst) begin
    if(!nrst)
        spill_reg <= 'h0;
    else if(!ready_i & slice_full)
        spill_reg <= slice_reg;
end

always @(posedge clk or negedge nrst) begin
    if(!nrst)
        spill_full <= 'h0;
    else if(!ready_i || spill_full)
        spill_full <= !ready_i & slice_full;
end

assign valid_o = slice_full || spill_full;
assign data_o = spill_full ? spill_reg : slice_reg;
assign ready_o = !spill_full;

endmodule

反压

出现反压有两个场景，

  1. 入口流量大于出口流量，即多对一。
  2. 在流水线中，后级还没有准备好，前级对后级进行数据冲刷，所以需要反压让前级数据保持不变；

在实际设计中遇到的情况：第一个对应的仲裁器，在icache和dcache同时读取SRAM数据时，需要以dcache优先，保证取数据优先，这时就要反压icache的访问内存行为，即拉低arready和rvalid；第二个对应在CPU五级流水线设计过程中，如果访存阶段通过axi协议读写内存数据时，需要将流水线停顿下来，发出反压信号stall，让前级流水线的寄存器保持不变。

不同场景的反压方法

  • 不带存储体的反压

    反压信号需要对流水线的所有寄存器进行控制；

eg. 下面加法问题，要求不使用存储器FIFO，完成握手设计；

思路：没有FIFO缓冲，相当于每级流水线的FIFO深度为1；逐级反压；

module handshake_pb (
    input  wire         clk,
    input  wire         rst,
    input  wire         valid_i,
    output wire         ready_o,
    input  wire  [31:0] a,
    input  wire  [31:0] b,
    input  wire  [31:0] c,
    input  wire  [31:0] d,
    input  wire  [31:0] e,
    input  wire  [31:0] f,
    output wire  [31:0] dout,
    input  wire         ready_i,
    output reg          valid_o
    );

    assign ready_o = ~valid_r1 || ready_r1;
    //pre_fetch结构
    //valid_r1为0 代表下一级无数据
    //ready_r1 	  代表下一级准备好了读

    always_ff @ (posedge clk) begin 
        if(rst)begin
            valid_r1    <= 1'b0;
        end
        else if(valid_i & ready_o)begin
            valid_r1    <= valid_i;
        end
    end
    
    always_ff @ (posedge clk) begin
        if(ready_o & valid_i)begin
            r1_ab       <= a + b;   //数据信号不复位
        end
    end

    assign ready_r1 = ~valid_r2 || ready_r2;

    reg [31 : 0] r2_abcd;

    always_ff @ (posedge clk) begin 
        if(rst)begin
            valid_r2    <= 1'b0;
        end
        else if(ready_r1 & valid_r1)begin
            valid_r2    <= valid_r1;
        end
    end
    
    always_ff @ (posedge clk) begin
        if(ready_r1 & valid_r1)begin
            r2_abcd     <= r1_ab + r1_cd;
        end
    end

    assign ready_r2 = ~valid_r3 || ready_i;
    
    reg [31 : 0] r3;

    always_ff @ (posedge clk) begin 
        if(rst)begin
            valid_r3    <= 1'b0;
        end
        else if(ready_r2 & valid_r2)begin
            valid_r3    <= valid_r2;
        end
    end
    
    always_ff @ (posedge clk) begin
        if(ready_r2 & valid_r2)begin
            r3          <= r2_ef + r2_abcd;
        end
    end

    assign dout     = r3;
    assign valid_o  = valid_r3;
endmodule

  • 带存储体的反压

    在握手模块中添加存储体，比如RAM和FIFO；将无法及时响应的数据缓冲到FIFO里，需要设置水线water line，也就是当waterline为多少时开始反压前一级模块。

​ 当FIFO数据个数达到阈值后，需要将ready_o拉低，告知上游模块，无法处理新的数据，完成反压。waterline计算：FIFO的深度 - 流水线的级数。上图水线为24

eg. 设计一个并行6输入32比特加法器，输出1个带截断的32比特加法结果，要求用三级流水设计，求带存储器FIFO，可以缓冲数据，带前后反压。

输入：
	1. 6个32bit数据 a，b，c，d，e，f
	2. 上一级的valid_i
	3. 下一级的ready_i
输出：
	1. 1个32bit结果 dout
	2. 给上一级的ready_o
	3. 给下一级的valid_o
module handshake_fifo #(
    parameter           FIFO_DATA_WIDTH = 32,
    parameter           FIFO_DEPTH = 8
)(
    input  wire         clk,
    input  wire         rst,
    input  wire         valid_i,
    input  wire         ready_i,
    input  wire  [31:0] a,
    input  wire  [31:0] b,
    input  wire  [31:0] c,
    input  wire  [31:0] d,
    input  wire  [31:0] e,
    input  wire  [31:0] f,
    output logic [31:0] dout,
    output logic        ready_o,
    output logic        valid_o
    );
    
    localparam          WATERLINE = FIFO_DEPTH - 3; //three levels' pipeline
    logic               handshake;
    logic               handshake_ff1;
    logic               handshake_ff2;
    logic               wr_en;
        
    assign handshake = ready_o & valid_i;

    always @ (posedge clk or posedge rst) begin
        if(rst) begin
            handshake_ff1 <= '0;
            handshake_ff2 <= '0;
        end
        else begin
            handshake_ff1 <= handshake;
            handshake_ff2 <= handshake_ff1;
        end
    end
    
    reg [31 : 0] r1_ab;
    always @ (posedge clk or posedge rst) begin
        if(rst) begin
            r1_ab <= '0;
        end
        else if(handshake)begin
            r1_ab <= a + b;
        end
    end
    
    reg [31 : 0] r1_cd;
    always @ (posedge clk or posedge rst) begin
        if(rst) begin
            r1_cd <= '0;
        end
        else if(handshake)begin
            r1_cd <= c + d;
        end
    end
    
    reg [31 : 0] r1_ef;
    always @ (posedge clk or posedge rst) begin
        if(rst) begin
            r1_ef <= '0;
        end
        else if(handshake)begin
            r1_ef <= e + f;
        end
    end
    
    reg [31 : 0] r2_abcd;
    always @ (posedge clk or posedge rst) begin
        if(rst) begin
            r2_abcd <= '0;
        end
        else if(handshake_ff1) begin
            r2_abcd <= r1_ab + r1_cd;
        end
    end
    
    reg [31 : 0] r2_ef;
    always @ (posedge clk or posedge rst) begin
        if(rst) begin
            r2_ef <= '0;
        end
        else if(handshake_ff1) begin
            r2_ef <= r1_ef;
        end
    end
    
    reg [31 : 0] r3;
    always @ (posedge clk or posedge rst) begin
        if(rst) begin
            r3 <= '0;
        end
        else if(handshake_ff2) begin
            r3 <= r2_ef + r2_abcd;
        end
    end
    
    always @ (posedge clk or posedge rst) begin
        if(rst) begin
            wr_en <= 1'b0;
        end
        else if(handshake_ff2) begin
            wr_en <= 1'b1;
        end
        else begin
            wr_en <= 1'b0;
        end
    end

    always_ff @(posedge clk)begin
        if(rst)begin
            ready_o <= 1'b0;
        end
        else if(usedw > WATERLINE)begin
            ready_o <= 1'b0;
        end
        else begin
            ready_o <= 1'b1;
        end
    end

    assign valid_o = ~empty;
     sync_fifo # (
        .MEM_TYPE   ("auto"         ),
        .READ_MODE  ("fwft"         ),
        .WIDTH      (FIFO_DATA_WIDTH),
        .DEPTH      (FIFO_DEPTH     )
    )fifo_inst(
        .clk    (clk                ), // input  wire
        .rst_n  (rst_n              ), // input  wire
        .wren   (wr_en              ), // input  wire
        .din    (r3                 ), // input  wire [WIDTH-1:0]
        .rden   (ready_i&!empty     ), // input  wire
        .dout   (dout               ), // output reg  [WIDTH-1:0]
        .empty  (empty              ), // output wire
        .usedw  (usedw              )
    );

endmodule
  • 逐级反压与跨级反压

​ 带多个存储体的反压分为两种：逐级反压和跨级反压；区别如下

​ 逐级反压：module3达到水线反压module2；module2达到水线反压module1；module1达到水线反压上游接口；每级都有握手信号需要处理

​ 跨级反压：module3的存储体深度 = waterlie3 + 在途1 + waterline1 + 在途2 + waterline2 + 在途3；只需要处理上游模块和module3的握手信号
