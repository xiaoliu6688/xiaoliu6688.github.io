Cache实现

根据上述的理论知识，确定cache的设计要求如下

  1. 采用直接映射缓存的方式，cache size为2KB，cache line为8Bytes，

  2. 当访问cache缺省时，采用读分配Read allocate和写分配Write allocate策略，

  3. cache的更新策略为写回Write Back

Icache实现

​ icache在流水线中的位置以及接口信号如下

​ 包含两组握手信号，第一个是ifu和PC寄存器的握手，只有ifpc_valid & ifpc_ready=1握手后，才可以更新ifu阶段的取值地址pc_i。第二个是ifu和icache的握手，只有cache_rdy & cache_vld 握手成功后，ifu阶段才可以拿到pc对应的指令，并通过inst_valid信号指示，给到IF_ID寄存器。Icache和SRAM通过AXI协议完成读取指令。

​ 由于icache只有对内存读取数据的需求，所以AXI协议中的写地址、写数据和写响应通道可以忽略，icache的整个状态跳转如下

assign allocate = state == ALLOCATE;
assign wstrb = (state == ALLOCATE) ? 8'b11111111 : 8'b00000000;
assign wdata = (state == ALLOCATE) ? ram_data : 64'h0;
assign ready = hit && (state == IDLE);
always @(*) begin
    case(state)
        IDLE : if((!hit) && valid)            next_state = RADDR;
                else                          next_state = IDLE;
        RADDR: if(axi_arvalid && axi_arready) next_state = RDATA;
                else                          next_state = RADDR;
        RDATA: if(axi_rvalid && axi_rready)   next_state = ALLOCATE;
                else                          next_state = RDATA;
        ALLOCATE: next_state = IDLE; 
        default : next_state = IDLE;
    endcase
end

只有在IDLE状态才会读取cache的数据，如果hit，直接输出有效的指令；如果miss，则通过AXI读取内存，首先跳转到RADDR状态，进行读地址通道的握手，然后跳转到读数据状态。读地址通道握手后，得到指令数据，然后跳转到读分配状态，将读出的指令加载到cachemem中。最终在IDLE状态，hit该cacheline，读出数据。

cachemem中对于hit的判断和cacheline的更新如下

always @(posedge clk or negedge rstn) begin
  if(!rstn) begin
  end else if(flash) begin
    for(i=0;i<DATA_SIZE;i=i+1) 
      v[i]	 <= 'h0;
      d[i]   <= 'h0;
      tag[i] <= 'h0;
      mem[i] <= 'h0;
  end else if(en & allocate) begin
      for(i=0;i<8;i=i+1) begin
      if(wen[i])
        mem[addr_index][8*i+:8] <= wdata[8*i+:8];
    end
    d[addr_index] <= 0;
    v[addr_index] <= 1;
    tag[addr_index] <= addr_tag; 
  end
end
assign addr_index = addr[3+:8];
assign addr_tag = addr[63:3+8];
assign offset	= addr[2:0]; 
assign hit = en && (addr_tag == tag[addr_index]) && v[addr_index];
assign rdata = offset[2] ?
&#123;32'b0, mem[addr_index][63:32]&#125; : &#123;32'b0, mem[addr_index][31:0]&#125;;

由于inst指令是32bit的，cacheline是64bit，通过判断offset[2]确定读取命中cacheline的高32bit还是低32bit。

在cache miss时，通过AXI访问内存时，通常需要多个周期。这个过程中，需要将流水线停顿下来，发出stall信号给PC寄存器，输出给IF_ID寄存器为nop指令。

Dcache实现

​ dcache在流水线中的位置以及接口信号如下

​ 只包含一组握手信号，lus和dcache之间；对dcache进行读操作的过程跟icache差不多。
​ 最大区别在于写操作，写缺省时，执行写分配，将写入地址的数据读取到cachemem中(这个过程和读分配差不多)，在IDLE状态将待更新的数据写入到cacheline中，并将对应的dirty位置1。在替换该cacheline时，判断dirty位为1，需要先将数据写入到内存中；因此，相比icache，多了很多状态。

always @(*) begin
    case (state)
        IDLE : next_state = (!hit && dirty && valid) ? WRITEBACK :
                            (!hit && valid) ? RADDR : IDLE;
        WRITEBACK : next_state = WADDR;
        ALLOCATE : next_state = IDLE;
        RADDR : if (axi_arvalid && axi_arready) next_state = RDATA;
        RDATA : if (axi_rvalid  && axi_rready ) next_state = ALLOCATE;
        WADDR : if (axi_awvalid && axi_awready) next_state = WDATA;
        WDATA : if (axi_wvalid  && axi_wready && axi_wlast ) next_state = WRESP;
        WRESP : if (axi_bvalid  && axi_bready ) next_state = RADDR;
        default : next_state = IDLE;
    endcase
end

  • 针对读写指令是字节/半字/字/双字

    在读写cacheline时需要根据地址的后三位使用offset来限制，保证读出的数据都是对齐的，写入数据对cacheline的数据更新也是对齐的。

    assign offset = addr[2:0];
    // Read
    always @* begin
      case(offset) 
        3'b000: rdata = mem[addr_index];
        3'b001: rdata = &#123;8'b0, mem[addr_index][63:8]&#125;;
        3'b010: rdata = &#123;16'b0, mem[addr_index][63:16]&#125;;
        3'b011: rdata = &#123;24'b0, mem[addr_index][63:24]&#125;;
        3'b100: rdata = &#123;32'b0, mem[addr_index][63:32]&#125;;
        3'b101: rdata = &#123;40'b0, mem[addr_index][63:40]&#125;;
        3'b110: rdata = &#123;48'b0, mem[addr_index][63:48]&#125;;
        3'b111: rdata = &#123;56'b0, mem[addr_index][63:56]&#125;;
        default: rdata = mem[addr_index];
      endcase
    end

    reg [7:0] we_offset;
    always @(*) begin
      case(offset)
        3'b000: we_offset = we << 0;
        3'b001: we_offset = we << 1;
        3'b010: we_offset = we << 2;
        3'b011: we_offset = we << 3;
        3'b100: we_offset = we << 4;
        3'b101: we_offset = we << 5;
        3'b110: we_offset = we << 6;
        3'b111: we_offset = we << 7;
        default : we_offset = we;
      endcase
    end

    reg [63:0] wdata_offset;
    always @(*) begin
      case(offset)
        3'b000: wdata_offset = wdata << 8*0;
        3'b001: wdata_offset = wdata << 8*1;
        3'b010: wdata_offset = wdata << 8*2;
        3'b011: wdata_offset = wdata << 8*3;
        3'b100: wdata_offset = wdata << 8*4;
        3'b101: wdata_offset = wdata << 8*5;
        3'b110: wdata_offset = wdata << 8*6;
        3'b111: wdata_offset = wdata << 8*7;
        default : wdata_offset = wdata;
      endcase
    end
    always @(posedge clk or negedge rstn) begin
        if (!rstn) begin
            for (i = 0; i < 8; i=i+1) begin
                v[i] = 0; d[i] = 0; tag[i] = 0;
                mem[i]=0;
            end
        end else if (en && hit) begin
          for (i = 0; i < 8; i=i+1) begin
            if (we_offset[i]) begin
              mem[addr_index][8 * i +: 8] <= wdata_offset[8 * i +: 8];
            end
          end
          d[addr_index] <= 1;
        end else if (en && allocate) begin
          for (i = 0; i < 8; i=i+1) begin
            if (we[i]) begin
              mem[addr_index][8 * i +: 8] <= wdata[8 * i +: 8];
            end
          end
          d[addr_index] <= 0;
          v[addr_index] <= 1;
          tag[addr_index] <= addr_tag;
        end
    end

在dcache读写内存时，需要多个周期，这时流水线需要停顿下来，发出stall信号，将前面各个阶段寄存器的数据保持不变。
